Jogo para tecprog.
Rodrigo e Ana.

jogo plataforma no estilo mario bros.

sessões:
personagem.
inimigos.
cenário.
física.

personagem:
mario.
Atributos
{
    básicos:
        anda para os lados.
        pula.
        uma vida.
    avançados:
        acelera baseado no tempo que o botão é pressionado.
        pula mais alto baseado no tempo que o botão é pressionado.
    futuros:
        power ups?
        mais de uma vida?
}
inimigos:
gomba.
Atributos
{
    básicos:
        anda automaticamente.
        uma vida.
}

koopa.
Atributos
{
    básicos:
        anda automaticamente.
        uma vida.
    avançados:
        duas vidas, sendo a segunda recolhido dentro do casco ficando parado.
        regenera a primeira vida após x segundos no casco.
    futuros:
        casco ser lançado pelo jogador?
}

cenário:
Atributos:
{
    básicos:
        chão.
        chão com abismos em que o jogador tem que saltar no tempo correto.
    avançados:
        obstáculos.
    futuros:
        subfase(cano)?
}

física:
Atributos:
{
    básicos:
        gravidade.
        colisão:
        entre inimigos:
            inversão do movimento.
        entre adversários:
            combate.
    avançados:
        combate:
            jogador encostar por cima->inimigo perde vida.
            jogador encostar pelo lado->jogador perde vida.
            jogador caindo na hora de saltar de um chão para o outro -> jogador perde vida.
}

Chefão
Atributos:
{
    básicos:
      mais vida.
      em tamanho, maior que o personagem.
      presente no fim da fase.
}
